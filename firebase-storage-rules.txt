// Reglas de Firebase Storage para Papiro Digital
// Estas reglas deben copiarse en Firebase Console → Storage → Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Regla para archivos de usuarios autenticados
    match /proyectos/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Regla para archivos públicos (portadas, etc.)
    match /public/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // Regla para covers/portadas de usuarios
    match /users/{userId}/covers/{allPaths=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Permitir lectura de archivos para usuarios autenticados
    match /{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
  }
}
